Core Layer

HeadlessRunner

The foundation. Wraps an Engine instance and drives the game loop without a browser.

let mut runner = HeadlessRunner::new(480, 720);
let result = runner.run(setup_fn, update_fn, render_fn, 120);

// With per-frame callback (receives frame number and dt)
let result = runner.run_with_frame_cb(
    |engine| { /* setup */ },
    |engine, frame, dt| { /* per-frame logic */ },
    120,
);

new(width: u32, height: u32) → Self

run(setup, update, render, frames) → SimResult

run_with_frame_cb(setup, per_frame, frames) → SimResult

SimResult

Universal output of every simulation run.

pub struct SimResult {
    pub frames_run: u64,
    pub final_metrics: FrameMetrics,
    pub game_state: HashMap<String, StateValue>,
    pub framebuffer_hash: u64,        // FNV-1a hash
    pub elapsed_sim_time: f64,
}

get_f64(key: &str) → Option<f64> Convenience accessor for numeric state values

pub enum ScheduledAction {

frame() → u64 Get the frame number

coords() → (f64, f64) Get the (x, y) coordinates

GameScenario

Declarative test: setup + actions + frame count + assertions = structured result.

let scenario = GameScenario {
    name: "my test".into(),
    width: 480, height: 720,
    setup_fn, update_fn, render_fn, action_dispatch,
    actions: vec![...],
    total_frames: 120,
    assertions: vec![
        Assertion::StateEquals { key: "score".into(), expected: 1.0, tolerance: 0.1 },
    ],
};
let result = scenario.run();
assert!(result.all_passed(), "{}", result.failure_report());

Assertion Types

VariantDescription
StateEquals { key, expected, tolerance }f64 value equals expected (±tolerance)
StateInRange { key, min, max }f64 value within [min, max]
StateGreaterThan { key, threshold }f64 value > threshold
StateLessThan { key, threshold }f64 value < threshold
FramebufferHash { expected }Exact framebuffer hash match
FramebufferChanged { previous }Framebuffer hash differs from previous

let builder = ScenarioBuilder::new(setup_fn, update_fn, render_fn, dispatch_fn);

ShotBuilder

High-level builder for slingshot-style input sequences. Angle: 0=right, 90=down, 180=left, 270=up.

let (actions, total_frames) = ShotBuilder::new()
    .with_drag_scale(120.0)      // pixels per power=1.0
    .with_settle_frames(180)     // 3 seconds settle time
    .aim_and_shoot(240.0, 500.0, 270.0, 0.8)
    .wait(60)
    .build();

Analysis Layer

run_sweep()

Run the same scenario with different parameter configurations.

let configs = vec![
    SweepConfig { label: "low".into(), overrides: vec![("drag".into(), 1.0)] },
    SweepConfig { label: "high".into(), overrides: vec![("drag".into(), 3.0)] },
];
let report = run_sweep(setup_fn, update_fn, render_fn, dispatch_fn,
    &actions, &configs, 120);

report.min_by_state("dist_to_hole");  // best config
report.summary_with_keys(&["dist_to_hole", "ball_x"]);

FitnessEvaluator

let evaluator = FitnessEvaluator::new()
    .add("proximity", 3.0, |sim| {
        score_distance("ball_x", "ball_y", "hole_x", "hole_y", 300.0, sim)
    })
    .add("completion", 2.0, |sim| {
        score_state_match("in_hole", 1.0, 0.5, sim)
    });

let fitness = evaluator.evaluate(&sim_result);
// fitness.grade() => "A+", "A", "B", "C", "D", "F"
// fitness.summary() => "total=0.85 | proximity=0.90(x3) completion=0.75(x2)"

Scoring helpers:

  • score_distance(x1, y1, x2, y2, max_dist, sim) — 1.0 at distance 0, 0.0 at max_dist
  • score_state_match(key, target, tol, sim) — 1.0 if value equals target
  • score_ratio(key, target, sim) — target/actual, capped at 1.0

RegressionSuite

let suite = RegressionSuite::new(&["ball_x", "stroke_count"])

Replay & record_replay()

let replay = record_replay("baseline",
    setup_fn, update_fn, render_fn, dispatch_fn,
    &actions, 120, &["ball_x", "ball_y"]);

replay.series("ball_x");           // Vec<f64> all frames
replay.get(30, "ball_x");          // Option<f64> at frame 30
replay.at(30);                     // Option<&ReplayFrame>
replay.first_frame_where("ball_vy", |vy| vy >= 0.0);
replay.len();
replay.is_empty();

compare_replays()

let cmp = compare_replays(&replay_a, &replay_b, &["ball_x", "ball_y"], 0.01);
cmp.is_identical(0.01);                    // all keys match?
let diff = cmp.key_diff("ball_x").unwrap();
// diff.max_delta, diff.max_delta_frame, diff.mean_delta, diff.first_divergence
cmp.first_visual_divergence;               // framebuffer hash divergence

Optimization Layer

Experiment

let result = Experiment::new("test", setup_fn, update_fn, render_fn, dispatch_fn)
    .with_actions(actions)
    .with_configs(configs)
    .with_frames(120)
    .with_fitness(evaluator)
    .with_baseline(baseline, &["ball_x"])

result.regression_ok();  // no regressions?
result.summary();

    .with_param(ParamRange::new("restitution", 0.1, 0.9, 0.1))

result.improved();       // fitness improved?
result.evaluations;      // total sims run
result.summary();

Orchestration Layer

Strategy

let result = Strategy::new("physics check", setup_fn, update_fn, render_fn, dispatch_fn)
    .record("baseline", vec![], 120, &["ball_x", "ball_y"])
    .record("with_shot", actions, 120, &["ball_x", "ball_y"])
    .compare("baseline", "with_shot", &["ball_x"], 0.01)
    .detect_anomalies("with_shot", &["ball_x", "ball_y"], 50.0, 30)
    .assert_state("with_shot", "stroke_count", StatePredicate::GreaterThan(0.0))
    .run();

result.all_passed();
result.failures();
result.summary();

Predicates: GreaterThan(f64), LessThan(f64), Equals(expected, tolerance), InRange(min, max)

TestHarness

let report = TestHarness::new(setup_fn, update_fn, render_fn, dispatch_fn)
    .add("idle", vec![], 60, vec![...assertions...])
    .add_with_fitness("quality", actions, 120, vec![], evaluator)
    .run();

report.all_passed();
report.passed();           // count
report.failed();           // count
report.avg_fitness();      // Option<f64>
report.summary();

GoldenTest

let test = GoldenTest::new(setup_fn, update_fn, render_fn, dispatch_fn)
    .with_actions(actions)
    .with_frames(120)
    .with_keys(&["ball_x", "ball_y"])
    .with_tolerance(0.01);

// Record reference
let golden = test.record_golden("v1");

// Compare against reference
let result = test.compare_against(&golden);
assert!(result.matches, "{}", result.summary());

Game Integration Contract

Every module takes 4 function pointers. Games provide:

fn setup(engine: &mut Engine)                              // Initialize state
fn update(engine: &mut Engine, dt: f64)                     // Game logic
fn render(engine: &mut Engine)                              // Draw to framebuffer
fn action_dispatch(engine: &mut Engine, a: &ScheduledAction) // Route inputs

S-League provides: sleague::setup_fight_only, sleague::update, sleague::render, sleague::dispatch_action
